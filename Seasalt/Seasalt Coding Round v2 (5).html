def read_text(file_path):
    extract=open(file_path,'r',encoding='utf-8')
    extracted=extract.read()
    extract.close()
    return extracted
def clean_line(line):
    clean_line = re.sub(' +',' ',line)
    if clean_line[0] == ' ':
        clean_line = clean_line[1:]
    return clean_line
def strip_ends(text_inp,ends):
    text_op=[]
    for text_line, ends in zip(text_inp,ends):
        text_op.append(text_line[ends[0]+1:ends[1]])
    return text_op

# serialize model to JSON
def save_model(model,name_suffix=''):
    model_json = model.to_json()
    model_name='model'+name_suffix
    json_filename=model_name+'.json'
    hdf5_filename=model_name+'.h5'
    with open(json_filename, "w") as json_file:
        json_file.write(model_json)
    # serialize weights to HDF5
    model.save_weights(hdf5_filename)
    print("Saved model to disk")
In [21]:
def reload_model(name_suffix):
    model_name='model'+name_suffix
    json_filename=model_name+'.json'
    hdf5_filename=model_name+'.h5'
    json_file = open(json_filename, 'r')
    loaded_model_json = json_file.read()
    json_file.close()
    loaded_model = model_from_json(loaded_model_json)
    # load weights into new model
    loaded_model.load_weights(hdf5_filename)
    print("Loaded model from disk")
    return loaded_model

4.3 Architecture:
The architecture I am intend to use here is an LSTM based encoder - decoder one.
The encoder with learn the context of the input sequences and pass those hidden states to the decoder.
The decoder is set to predict the next character of the target sequence.

def onehot_encode_array(np_arr):
    flat_arr=[item for seq in np_arr for item in seq]
    len_enc=max(flat_arr)+1
    encoded_matrix=np.zeros((np_arr.shape[0],np_arr.shape[1],len_enc))
    for i in range(np_arr.shape[0]):
        for j in range(np_arr.shape[1]):
            encoded_matrix[i,j,np_arr[i,j]]=1
    return encoded_matrix
